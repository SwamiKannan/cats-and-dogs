def filter_photos(k=[]):
    if k:
        filters=[]
        while len(filters)<3:
            pick=random.choice(k)
            if pick not in filters:
                filters.append(pick)
        #print(filters)
        transform_filters=transforms.Compose([f for f in (filters+[transforms.Resize(224),transforms.CenterCrop(224),transforms.ToTensor()])])
    else:
        transform_filters=transforms.Compose([transforms.Resize(224),transforms.CenterCrop(224),transforms.ToTensor()])
    return transform_filters

def convert(file_names,k=None):
    #print('base:','..\\Data\\CATS_DOGS\\train\\DOG\\14.jpg')
    save_path='..\\Data\\CATS_DOGS\\train1'

    for name in file_names:
        if '..\\Data\\CATS_DOGS\\train\\' in name[0]:
            i=Image.open(name[0],mode='r').convert('RGB')
            ##print(name[0])
            if i.mode != "RGB":
                i.convert("RGBA")
                #print(i.mode)
            transform_object=filter_photos(k)
            tr=transform_object(i)
            PIL_img=transforms.functional.to_pil_image(tr)
            if '..\\Data\\CATS_DOGS\\train\\CAT\\' in name[0]:
                if k:
                    save_img_path=save_path+'\\CAT\\'+name[1][:-4]+'_res.jpg'
                else:
                    save_img_path=save_path+'\\CAT\\'+name[1]
                #print(save_img_path)
                PIL_img.save(save_img_path, "JPEG")
                PIL_img.close()
            else:
                if k:
                    save_img_path=save_path+'\\DOG\\'+name[1][:-4]+'_res.jpg'
                else:
                    save_img_path=save_path+'\\DOG\\'+name[1]
                #print(save_img_path)
                PIL_img.save(save_img_path, "JPEG")
                PIL_img.close()
